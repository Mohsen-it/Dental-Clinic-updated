# شرح نظام معرف الجهاز (Hardware ID - HWID)

## نظرة عامة

معرف الجهاز هو الرقم الفريد الذي يربط الترخيص بجهاز معين. يعتمد النظام على:
- `node-machine-id` للحصول على معرف الجهاز من النظام
- `SHA-256` لإنشاء hash فريد ومشفر
- `Salt` مخصص لكل تطبيق لمنع التصادم

## 1. كيف يعمل معرف الجهاز

### 1.1 الحصول على معرف الجهاز

```javascript
// في licenseManager.ts
import { machineIdSync } from 'node-machine-id'

function generateHWID() {
  try {
    // الخطوة 1: الحصول على machine ID من النظام
    const machineId = machineIdSync()
    
    // الخطوة 2: إنشاء hash مشفر
    const hash = createHash('sha256')
    hash.update(machineId + APP_SALT) // إضافة salt فريد
    
    // الخطوة 3: إرجاع أول 32 حرف
    return hash.digest('hex').substring(0, 32)
  } catch (error) {
    console.error('Error generating HWID:', error)
    
    // Fallback: استخدام معلومات النظام
    const fallbackData = `${process.platform}-${process.arch}-${Date.now()}`
    const hash = createHash('sha256')
    hash.update(fallbackData + APP_SALT)
    return hash.digest('hex').substring(0, 32)
  }
}
```

### 3.1 سكريبت Node.js

```javascript
// scripts/showDeviceId.js
/**
 * سكريبت لعرض معرف الجهاز
 * التشغيل: node scripts/showDeviceId.js
 */

const crypto = require('crypto')

// نفس APP_SALT المستخدم في licenseManager
const APP_SALT = 'dental-clinic-license-salt-2025'

function getMachineId() {
  try {
    const { machineIdSync } = require('node-machine-id')
    return machineIdSync()
  } catch (error) {
    console.error('node-machine-id not available:', error.message)
    return null
  }
}

function generateHWID(machineId) {
  if (!machineId) {
    // Fallback
    const fallbackData = `${process.platform}-${process.arch}-${Date.now()}`
    const hash = crypto.createHash('sha256')
    hash.update(fallbackData + APP_SALT)
    return {
      hwid: hash.digest('hex').substring(0, 32),
      source: 'fallback',
      data: fallbackData
    }
  }
  
  const hash = crypto.createHash('sha256')
  hash.update(machineId + APP_SALT)
  
  return {
    hwid: hash.digest('hex').substring(0, 32),
    source: 'machine-id',
    machineId: machineId
  }
}
```

## 6. التحقق من HWID في التحقق من الترخيص

### 6.1 التحقق عند التفعيل

```javascript
// في licenseManager.js
async function activateLicense(licenseKey) {
  // 1. التحقق من تنسيق المفتاح
  if (!validateLicenseFormat(licenseKey)) {
    return { isValid: false, error: 'Invalid license format' }
  }

  // 2. الحصول على HWID الحالي
  const currentHWID = generateHWID()

  // 3. التحقق من المفتاح
  const isValid = validateKey(licenseKey, currentHWID)

  if (isValid) {
    // 4. حفظ الترخيص
    const licenseData = {
      license: licenseKey,
      hwid: currentHWID,  // حفظ HWID مع الترخيص
      timestamp: Date.now(),
      activated: true
    }

    await storeLicense(licenseData)

    return {
      isValid: true,
      licenseData: licenseData
    }
  } else {
    return {
      isValid: false,
      error: 'License is not valid for this device'
    }
  }
}
```

### 6.2 التحقق عند التشغيل

```javascript
// في licenseManager.js
async function validateStoredLicense() {
  try {
    // 1. قراءة البيانات المشفرة
    const encryptedData = licenseStore.get('licenseData')
    if (!encryptedData) {
      return { isValid: false, error: 'No license found' }
    }

    // 2. فك التشفير
    const licenseData = decryptLicenseData(encryptedData)

    // 3. الحصول على HWID الحالي
    const currentHWID = generateHWID()

    // 4. التحقق من التطابق
    if (licenseData.hwid !== currentHWID) {
      return {
        isValid: false,
        error: 'License is bound to a different device'
      }
    }

    // 5. التحقق من التفعيل
    if (!licenseData.activated) {
      return {
        isValid: false,
        error: 'License is not activated'
      }
    }

    return {
      isValid: true,
      licenseData: licenseData
    }

  } catch (error) {
    return {
      isValid: false,
      error: 'License validation failed'
    }
  }
}
```

## 10. ملف getDeviceId.js الكامل

```javascript
/**
 * getDeviceId.js - سكريبت الحصول على معرف الجهاز
 * 
 * الوظائف:
 * 1. الحصول على معرف الجهاز الفريد
 * 2. إنشاء مفتاح ترخيص للجهاز
 * 3. حفظ المفتاح في ملف
 * 
 * التشغيل:
 *   node scripts/getDeviceId.js
 *   node scripts/getDeviceId.js generate
 *   node scripts/getDeviceId.js generate PROFESSIONAL GLOBAL
 */

const crypto = require('crypto')
const fs = require('fs')
const path = require('path')

// إعدادات النظام
const APP_NAME = 'Dental Clinic'
const APP_SALT = 'dental-clinic-license-salt-2025'
const MASTER_KEY = 'DENTAL_CLINIC_MASTER_KEY_2025_SECURE_ENCRYPTION'

/**
 * الحصول على معرف الجهاز من النظام
 */
function getMachineId() {
  try {
    const { machineIdSync } = require('node-machine-id')
    return machineIdSync()
  } catch (error) {
    console.warn('⚠️ node-machine-id not available, using fallback')
    return null
  }
}

/**
 * إنشاء HWID من معرف الجهاز
 */
function generateHWID(machineId) {
  if (!machineId) {
    // Fallback: استخدام معلومات النظام
    const fallbackData = `${process.platform}-${process.arch}-${Date.now()}`
    const hash = crypto.createHash('sha256')
    hash.update(fallbackData + APP_SALT)
    return {
      hwid: hash.digest('hex').substring(0, 32),
      source: 'fallback',
      data: fallbackData
    }
  }

  const hash = crypto.createHash('sha256')
  hash.update(machineId + APP_SALT)

  return {
    hwid: hash.digest('hex').substring(0, 32),
    source: 'machine-id',
    machineId: machineId
  }
}

/**
 * إنشاء مفتاح ترخيص خوارزمي
 */
function generateLicenseKey(deviceId, licenseType = 'PROFESSIONAL') {
  const seed = deviceId + MASTER_KEY + licenseType
  const baseHash = crypto.createHash('sha256').update(seed).digest('hex')

  const part1 = baseHash.substring(0, 5).toUpperCase()
  const part2 = baseHash.substring(8, 13).toUpperCase()
  const part3 = baseHash.substring(16, 21).toUpperCase()
  const part4 = baseHash.substring(24, 29).toUpperCase()

  return `${part1}-${part2}-${part3}-${part4}`
}

/**
 * التحقق من مفتاح ترخيص
 */
function validateLicenseKey(licenseKey, deviceId) {
  const licenseTypes = ['STANDARD', 'PROFESSIONAL', 'ENTERPRISE', 'PREMIUM', 'ULTIMATE']

  for (const type of licenseTypes) {
    const expectedKey = generateLicenseKey(deviceId, type)
    if (expectedKey === licenseKey) {
      return { valid: true, type: type }
    }
  }

  return { valid: false }
}

/**
 * حفظ بيانات الترخيص
 */
function saveLicenseData(deviceId, licenseKey, metadata = {}) {
  const data = {
    deviceId: deviceId,
    licenseKey: licenseKey,
    metadata: {
      licenseType: metadata.type || 'PROFESSIONAL',
      region: metadata.region || 'GLOBAL',
      isLifetime: true,
      maxDevices: 1,
      createdAt: new Date().toISOString(),
      ...metadata
    }
  }

  const filename = `device-license-${deviceId.substring(0, 8)}.json`
  const filepath = path.join(__dirname, '..', filename)

  fs.writeFileSync(filepath, JSON.stringify(data, null, 2), 'utf8')

  return { filepath, data }
}
```
