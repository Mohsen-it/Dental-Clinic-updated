# نظام التراخيص الكامل - Dental Clinic Management System

## نظرة عامة شاملة

نظام الترخيص هو نظام متكامل لحماية تطبيق Electron من الوصول غير المصرح به. يعتمد على ربط الترخيص بمعرف الجهاز الفريد (Hardware ID) وتشفير البيانات باستخدام AES-256-GCM.

### 2.2 مولد المفاتيح المرتبطة بالجهاز (deviceBoundLicenseGenerator.js)

```javascript
class DeviceBoundLicenseGenerator {
  constructor() {
    this.masterKey = 'DENTAL_CLINIC_MASTER_KEY_2025_SECURE_ENCRYPTION'
    this.algorithm = 'aes-256-cbc'
  }

  // إنشاء مفتاح مرتبط بجهاز معين
  generateDeviceBoundLicense(deviceId, metadata = {}) {
    const licenseKey = this.generateAlgorithmicKey(deviceId, metadata)
    return {
      licenseKey: licenseKey,
      deviceId: deviceId,
      metadata: {
        licenseType: metadata.licenseType || 'STANDARD',
        region: metadata.region || 'GLOBAL',
        isLifetime: true,
        maxDevices: 1
      },
      generatedAt: new Date().toISOString()
    }
  }

  // إنشاء مفتاح خوارزمي
  generateAlgorithmicKey(deviceId, metadata = {}) {
    const seed = deviceId + this.masterKey + (metadata.licenseType || 'STANDARD')
    const baseHash = crypto.createHash('sha256').update(seed).digest('hex')
    
    const part1 = baseHash.substring(0, 5)
    const part2 = baseHash.substring(8, 13)
    const part3 = baseHash.substring(16, 21)
    const part4 = baseHash.substring(24, 29)
    
    return `${part1}-${part2}-${part3}-${part4}` // XXXXX-XXXXX-XXXXX-XXXXX
  }

  // التحقق من مفتاح مرتبط بالجهاز
  validateDeviceBoundLicense(licenseKey) {
    const currentDeviceId = this.getCurrentDeviceId()
    return this.validateAlgorithmicKey(licenseKey, currentDeviceId)
  }
}
```

### 2.3 مدقق مفاتيح الإنتاج (productionLicenseValidator.js)

```javascript
class ProductionLicenseValidator {
  constructor() {
    this.productionLicenses = null
    this.loadProductionLicenses()
  }

  // تحميل المفاتيح من ملف JSON
  loadProductionLicenses() {
    const licensePath = path.join(__dirname, '..', 'production-licenses.json')
    if (fs.existsSync(licensePath)) {
      const licenseData = JSON.parse(fs.readFileSync(licensePath, 'utf8'))
      this.productionLicenses = licenseData.licenses
      return true
    }
    return false
  }

  // التحقق من صحة مفتاح
  isValidLicense(licenseKey) {
    const normalizedKey = licenseKey.trim().toUpperCase()
    return this.productionLicenses.some(license => license.key === normalizedKey)
  }

  // الحصول على معلومات المفتاح
  getLicenseInfo(licenseKey) {
    const license = this.productionLicenses.find(l => l.key === normalizedKey)
    if (license) {
      return {
        key: license.key,
        id: license.id,
        hash: license.hash,
        metadata: license.metadata,
        isProduction: true
      }
    }
    return null
  }
}
```

### 2.4 متتبع المفاتيح المستخدمة (usedLicensesTracker.js)

```javascript
class UsedLicensesTracker {
  constructor() {
    this.usedLicensesFile = path.join(__dirname, '..', 'used-licenses.json')
    this.usedLicenses = this.loadUsedLicenses()
  }

  // إنشاء hash للمفتاح
  hashLicenseKey(licenseKey) {
    return createHash('sha256')
      .update(licenseKey + 'dental-clinic-salt-2025')
      .digest('hex')
  }

  // التحقق من توفر المفتاح
  isLicenseAvailable(licenseKey, currentHWID) {
    const hashedKey = this.hashLicenseKey(licenseKey)
    const usedLicense = this.usedLicenses[hashedKey]

    if (!usedLicense) return true // المفتاح غير مُستخدم
    if (usedLicense.hwid === currentHWID) return true // نفس الجهاز
    return false // مستخدم على جهاز آخر
  }

  // تسجيل استخدام مفتاح
  markLicenseAsUsed(licenseKey, hwid, additionalData = {}) {
    const hashedKey = this.hashLicenseKey(licenseKey)
    this.usedLicenses[hashedKey] = {
      hwid: hwid,
      activatedAt: new Date().toISOString(),
      activationCount: 1,
      ...additionalData
    }
    this.saveUsedLicenses()
  }

  // تحرير مفتاح (للدعم الفني)
  releaseLicense(licenseKey) {
    const hashedKey = this.hashLicenseKey(licenseKey)
    delete this.usedLicenses[hashedKey]
    this.saveUsedLicenses()
  }
}
```

## 3. واجهة المستخدم (LicenseEntryScreen.tsx)

```typescript
interface LicenseEntryScreenProps {
  onActivate: (licenseKey: string) => Promise<{ success: boolean; error?: string }>
  isLoading?: boolean
  machineInfo?: {
    hwid: string
    platform: string
    arch: string
  }
}

export default function LicenseEntryScreen({
  onActivate,
  isLoading = false,
  machineInfo
}: LicenseEntryScreenProps) {
  const [licenseKey, setLicenseKey] = useState('')
  const [error, setError] = useState('')
  const [isValidating, setIsValidating] = useState(false)

  // تنسيق المفتاح أثناء الكتابة
  const formatLicenseKey = (value: string) => {
    const cleaned = value.replace(/[^A-Z0-9]/gi, '').toUpperCase()
    const groups = []
    for (let i = 0; i < cleaned.length; i += 5) {
      groups.push(cleaned.slice(i, i + 5))
    }
    return groups.slice(0, 4).join('-')
  }

  const handleLicenseKeyChange = (e) => {
    const formatted = formatLicenseKey(e.target.value)
    setLicenseKey(formatted)
    setError('')
  }

  const validateLicenseFormat = (key) => {
    const regex = /^[A-Z0-9]{5}-[A-Z0-9]{5}-[A-Z0-9]{5}-[A-Z0-9]{5}$/
    return regex.test(key)
  }

  const handleSubmit = async (e) => {
    e.preventDefault()
    if (!validateLicenseFormat(licenseKey)) {
      setError('تنسيق مفتاح الترخيص غير صحيح')
      return
    }
    setIsValidating(true)
    const result = await onActivate(licenseKey)
    if (!result.success) {
      setError(result.error)
    }
    setIsValidating(false)
  }

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="text"
        value={licenseKey}
        onChange={handleLicenseKeyChange}
        placeholder="XXXXX-XXXXX-XXXXX-XXXXX"
        maxLength={23}
      />
      {/* عرض معرف الجهاز للنسخ */}
      {machineInfo && (
        <div>
          <code>{machineInfo.hwid}</code>
          <button onClick={() => navigator.clipboard.writeText(machineInfo.hwid)}>
            نسخ
          </button>
        </div>
      )}
      <button type="submit" disabled={isLoading || isValidating}>
        تفعيل الترخيص
      </button>
    </form>
  )
}
```

## 4. تكامل React (useLicense Hook)

```typescript
import { useState, useEffect, useCallback } from 'react'

export function useLicense() {
  const [licenseState, setLicenseState] = useState({
    isLicenseValid: false,
    isFirstRun: true,
    isLoading: true,
    error: null,
    licenseData: null,
    machineInfo: null
  })

  // التحقق من حالة الترخيص
  const checkLicenseStatus = useCallback(async () => {
    const result = await window.electronAPI.license.checkStatus()
    return result
  }, [])

  // الحصول على معلومات الجهاز
  const getMachineInfo = useCallback(async () => {
    const result = await window.electronAPI.license.getMachineInfo()
    return result
  }, [])

  // تفعيل الترخيص
  const activateLicense = useCallback(async (licenseKey: string) => {
    const result = await window.electronAPI.license.activate(licenseKey.trim())
    if (result.success) {
      await refreshLicenseStatus()
    }
    return result
  }, [])

  // تحديث حالة الترخيص
  const refreshLicenseStatus = useCallback(async () => {
    const [licenseStatus, machineInfo] = await Promise.all([
      checkLicenseStatus(),
      getMachineInfo()
    ])

    setLicenseState({
      isLicenseValid: licenseStatus.isValid,
      isFirstRun: licenseStatus.isFirstRun,
      isLoading: false,
      error: licenseStatus.error || null,
      licenseData: licenseStatus.licenseData,
      machineInfo: machineInfo
    })
  }, [checkLicenseStatus, getMachineInfo])

  // تهيئة عند التحميل
  useEffect(() => {
    refreshLicenseStatus()
  }, [refreshLicenseStatus])

  return {
    isLicenseValid: licenseState.isLicenseValid,
    isFirstRun: licenseState.isFirstRun,
    isLoading: licenseState.isLoading,
    error: licenseState.error,
    licenseData: licenseState.licenseData,
    machineInfo: licenseState.machineInfo,
    activateLicense,
    refreshLicenseStatus,
    clearLicenseData: () => window.electronAPI.license.clearData()
  }
}
```

## 12. خوارزمية التشفير

```javascript
// التشفير (AES-256-GCM)
function encryptLicenseData(data) {
  const key = pbkdf2Sync(APP_SALT, 'license-encryption', 100000, 32, 'sha256')
  const iv = randomBytes(16)
  const cipher = createCipheriv('aes-256-gcm', key, iv)
  
  const jsonData = JSON.stringify(data)
  let encrypted = cipher.update(jsonData, 'utf8', 'hex')
  encrypted += cipher.final('hex')
  
  const authTag = cipher.getAuthTag()
  
  // دمج IV و auth tag والبيانات المشفرة
  const result = iv.toString('hex') + ':' + authTag.toString('hex') + ':' + encrypted
  return Buffer.from(result).toString('base64')
}

// فك التشفير
function decryptLicenseData(encryptedData) {
  const key = pbkdf2Sync(APP_SALT, 'license-encryption', 100000, 32, 'sha256')
  const decodedData = Buffer.from(encryptedData, 'base64').toString('utf8')
  const [ivHex, authTagHex, encrypted] = decodedData.split(':')
  
  const iv = Buffer.from(ivHex, 'hex')
  const authTag = Buffer.from(authTagHex, 'hex')
  const decipher = createDecipheriv('aes-256-gcm', key, iv)
  decipher.setAuthTag(authTag)
  
  let decrypted = decipher.update(encrypted, 'hex', 'utf8')
  decrypted += decipher.final('utf8')
  
  return JSON.parse(decrypted)
}
```
